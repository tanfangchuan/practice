#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int i, m;
	while (cin >> m)
	{
		cout << m << " = ";
		i = 2;
		while (1)
		{
			if (m%i == 0)
			{
				m = m / i;
				cout << i;
				break;
			}
			i++;
		}
		i = 2;
		while (m != 1)
		{
			while (1)
			{
				if (m%i == 0)
				{
					m = m / i;
					break;
				}
				i++;
			}
			cout << " * " << i;
		}
		cout << endl;
	}
	system("pause");
	return 0;
}
